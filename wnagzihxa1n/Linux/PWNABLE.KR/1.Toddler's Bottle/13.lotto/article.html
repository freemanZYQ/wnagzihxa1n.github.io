大概就是一个游戏，输入一个6字节的数据，然后经过一个随机函数简单计算出来进行对比要有6字节是匹配的，也就是说6字节不能有重复的字节

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <fcntl.h>

unsigned char submit[6];

void play() {

    int i;
    printf("Submit your 6 lotto bytes : ");
    fflush(stdout);

    int r;
    r = read(0, submit, 6);

    printf("Lotto Start!\n");
    //sleep(1);

    // generate lotto numbers
    int fd = open("/dev/urandom", O_RDONLY);
    if(fd == -1) {
        printf("error. tell admin\n");
        exit(-1);
    }
    unsigned char lotto[6];
    if(read(fd, lotto, 6) != 6) {
        printf("error2. tell admin\n");
        exit(-1);
    }
    for(i = 0; i < 6; i++) {
        lotto[i] = (lotto[i] % 45) + 1;        // 1 ~ 45
    }
    close(fd);

    // calculate lotto score
    int match = 0, j = 0;
    for(i = 0; i < 6; i++) {
        for(j = 0; j < 6; j++) {
            if(lotto[i] == submit[j]) {
                match++;
            }
        }
    }
    // win!
    if(match == 6) {
        system("/bin/cat flag");
    }
    else {
        printf("bad luck...\n");
    }
}
